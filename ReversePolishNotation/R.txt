is_prime <- function(n) {
  if (n <= 1) {
    return(FALSE)
  } else if (n <= 3) {
    return(TRUE)
  } else if (n %% 2 == 0 || n %% 3 == 0) {
    return(FALSE)
  }
  
  i <- 5;
  while (i * i <= n) {
    if (n %% i == 0 || n %% (i + 2) == 0) {
      return(FALSE)
    }

    i <- i + 6;
  }

  return(TRUE)
}

# Цикл для вывода простых чисел до 100
for (i in 1:100) {
  if (is_prime(i)) {
    print(i)
  }
}
